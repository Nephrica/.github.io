document.getElementById("africaMap").addEventListener("load", function () { const svg = this.contentDocument; const clickableCodes = ["ET", "KE", "CD"]; const baseColor = "#d9d1e8"; // ✅ STEP 1: FORCE all paths to same base color const paths = svg.querySelectorAll("path"); paths.forEach(p => { p.style.fill = baseColor; p.style.stroke = "#ffffff"; p.style.strokeWidth = "1"; p.style.transition = "fill 0.2s ease"; // 🔑 THIS removes hardcoded SVG colors p.removeAttribute("fill"); }); // ✅ STEP 2: Add interactivity ONLY to clickable countries clickableCodes.forEach(code => { const el = svg.getElementById(code); if (!el) return; el.style.cursor = "pointer"; el.addEventListener("mouseenter", () => { if (!el.dataset.active) { el.style.fill = "#6a42b8"; } }); el.addEventListener("mouseleave", () => { if (!el.dataset.active) { el.style.fill = baseColor; } }); el.addEventListener("click", () => { clickableCodes.forEach(c => { const other = svg.getElementById(c); if (other) { other.dataset.active = ""; other.style.fill = baseColor; } }); el.dataset.active = "true"; el.style.fill = "#d24ca3"; renderMembers(code); }); }); });
